    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";
        import { getAuth, signInAnonymously } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";
        import { getFirestore, doc, setDoc, onSnapshot } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";

        const firebaseConfig = JSON.parse(__firebase_config);
        const app = initializeApp(firebaseConfig);
        const auth = getAuth(app);
        const db = getFirestore(app);
        const appId = typeof __app_id !== 'undefined' ? __app_id : 'pao-pao-march-8-ayauyllym';

        const boardElement = document.getElementById('game-board');
        const uploadInput = document.getElementById('image-upload');
        const canvas = document.getElementById('line-canvas');
        const ctx = canvas.getContext('2d');
        const timerText = document.getElementById('timer-text');
        const timerBar = document.getElementById('timer-bar');
        const winModal = document.getElementById('win-modal');
        const nextBtn = document.getElementById('next-btn');
        const levelDisplay = document.getElementById('level-display');

        let currentLevel = 1;
        let ROWS, COLS;
        let board = [];
        let selected = null;
        let customImages = [];
        let isProcessing = false;
        let gameActive = false;
        let timeTotal = 120;
        let timeRemaining = timeTotal;
        let timerId = null;
        let hintsLeft = 3;

        const defaultIcons = ['🌸', '🌷', '🌹', '🌺', '🌻', '🌼', '🐣', '🦋', '🎈', '🍓', '🍰', '💎'];

        async function startApp() {
            try {
                await signInAnonymously(auth);
                onSnapshot(doc(db, 'artifacts', appId, 'public', 'data', 'gallery'), (docSnap) => {
                    if (docSnap.exists()) customImages = docSnap.data().photos || [];
                    if (customImages.length === 0 || window.location.hash === '#admin') document.getElementById('admin-panel').classList.remove('hidden');
                    document.getElementById('loading').classList.add('hidden');
                    initGame();
                });
            } catch (err) {
                document.getElementById('loading').classList.add('hidden');
                initGame();
            }
        }

        async function compressImage(base64Str) {
            return new Promise((resolve) => {
                const img = new Image();
                img.src = base64Str;
                img.onload = () => {
                    const canvas = document.createElement('canvas');
                    canvas.width = 200; canvas.height = 200;
                    canvas.getContext('2d').drawImage(img, 0, 0, 200, 200);
                    resolve(canvas.toDataURL('image/jpeg', 0.8));
                };
            });
        }

        uploadInput.onchange = async (e) => {
            const files = Array.from(e.target.files).slice(0, 18);
            if (files.length < 4) return;
            const processed = [];
            for (let f of files) {
                const b64 = await new Promise(r => { const reader = new FileReader(); reader.onload = ev => r(ev.target.result); reader.readAsDataURL(f); });
                processed.push(await compressImage(b64));
            }
            await setDoc(doc(db, 'artifacts', appId, 'public', 'data', 'gallery'), { photos: processed });
        };

        function initGame() {
            if (currentLevel === 1) { ROWS = 4; COLS = 4; timeTotal = 120; }
            else if (currentLevel === 2) { ROWS = 6; COLS = 4; timeTotal = 120; }
            else if (currentLevel === 3) { ROWS = 6; COLS = 6; timeTotal = 180; }
            else { 
                ROWS = 8; COLS = 8; timeTotal = 300; 
                document.getElementById('antistress-label').classList.remove('hidden');
            }

            hintsLeft = 3;
            document.getElementById('hints-count').textContent = hintsLeft;
            levelDisplay.textContent = currentLevel;
            board = []; selected = null; gameActive = true; isProcessing = false;
            
            let pool = [];
            const pairsNeeded = (ROWS * COLS) / 2;
            const source = customImages.length >= 6 ? customImages : defaultIcons;
            for (let i = 0; i < pairsNeeded; i++) {
                const item = source[i % source.length];
                pool.push(item, item);
            }
            pool.sort(() => Math.random() - 0.5);

            for (let r = 0; r < ROWS + 2; r++) {
                board[r] = [];
                for (let c = 0; c < COLS + 2; c++) {
                    if (r === 0 || r === ROWS + 1 || c === 0 || c === COLS + 1) board[r][c] = null;
                    else board[r][c] = pool.pop();
                }
            }
            
            ensurePossibleMoves();
            renderBoard();
            startTimer();
        }

        function ensurePossibleMoves() {
            let attempts = 0;
            while (!checkAnyPossibleMove() && attempts < 50) {
                let tiles = [];
                for (let r = 1; r <= ROWS; r++) for (let c = 1; c <= COLS; c++) if (board[r][c]) tiles.push(board[r][c]);
                tiles.sort(() => Math.random() - 0.5);
                for (let r = 1; r <= ROWS; r++) for (let c = 1; c <= COLS; c++) if (board[r][c]) board[r][c] = tiles.pop();
                attempts++;
            }
        }

        function checkAnyPossibleMove() {
            const tilesPos = [];
            for (let r = 1; r <= ROWS; r++) for (let c = 1; c <= COLS; c++) if (board[r][c]) tilesPos.push({r, c, val: board[r][c]});
            for (let i = 0; i < tilesPos.length; i++) {
                for (let j = i + 1; j < tilesPos.length; j++) {
                    if (tilesPos[i].val === tilesPos[j].val && findPath(tilesPos[i].r, tilesPos[i].c, tilesPos[j].r, tilesPos[j].c)) 
                        return {t1: tilesPos[i], t2: tilesPos[j]};
                }
            }
            return false;
        }

        function renderBoard() {
            boardElement.innerHTML = '';
            boardElement.style.gridTemplateColumns = `repeat(${COLS}, 1fr)`;
            const isLevel4 = currentLevel === 4;
            for (let r = 1; r <= ROWS; r++) {
                for (let c = 1; c <= COLS; c++) {
                    const tile = document.createElement('div');
                    tile.dataset.r = r; tile.dataset.c = c;
                    tile.className = `tile flex items-center justify-center ${isLevel4 ? 'p-0.5' : ''}`;
                    if (board[r][c] === null) tile.classList.add('empty');
                    else {
                        const content = board[r][c];
                        if (content.startsWith('data:image')) {
                            const img = document.createElement('img');
                            img.src = content; img.className = 'w-full h-full object-cover rounded-md';
                            tile.appendChild(img);
                        } else {
                            tile.textContent = content; tile.classList.add(isLevel4 ? 'text-xl' : 'text-3xl');
                        }
                        tile.onclick = () => handleTileClick(r, c, tile);
                    }
                    boardElement.appendChild(tile);
                }
            }
            resizeCanvas();
        }

        function startTimer() {
            clearInterval(timerId);
            timeRemaining = timeTotal;
            updateTimerUI();
            timerId = setInterval(() => {
                if (!gameActive) return;
                timeRemaining--;
                updateTimerUI();
                if (timeRemaining <= 0) {
                    gameActive = false; clearInterval(timerId);
                    document.getElementById('lose-modal').classList.remove('hidden');
                }
            }, 1000);
        }

        function updateTimerUI() {
            const mins = Math.floor(timeRemaining / 60);
            const secs = timeRemaining % 60;
            timerText.textContent = `${mins}:${secs.toString().padStart(2, '0')}`;
            timerBar.style.width = `${(timeRemaining / timeTotal) * 100}%`;
        }

        function handleTileClick(r, c, el) {
            if (isProcessing || !gameActive || board[r][c] === null) return;
            if (selected) {
                if (selected.r === r && selected.c === c) {
                    el.classList.remove('selected'); selected = null; return;
                }
                if (board[r][c] === board[selected.r][selected.c]) {
                    const path = findPath(selected.r, selected.c, r, c);
                    if (path) {
                        drawConnectLine(path); isProcessing = true;
                        setTimeout(() => {
                            board[r][c] = null; board[selected.r][selected.c] = null;
                            selected = null; isProcessing = false;
                            clearCanvas(); 
                            if (checkWin()) handleWin();
                            else { ensurePossibleMoves(); renderBoard(); }
                        }, 400);
                        return;
                    }
                }
                document.querySelector(`[data-r="${selected.r}"][data-c="${selected.c}"]`)?.classList.remove('selected');
                selected = { r, c }; el.classList.add('selected');
            } else {
                selected = { r, c }; el.classList.add('selected');
            }
        }

        function findPath(r1, c1, r2, c2) {
            const queue = [{r: r1, c: c1, dir: -1, turns: 0, path: [{r: r1, c: c1}]}];
            const visited = new Map();
            while (queue.length > 0) {
                const cur = queue.shift();
                if (cur.r === r2 && cur.c === c2) return cur.path;
                const dirs = [[0,1], [0,-1], [1,0], [-1,0]];
                for (let i = 0; i < 4; i++) {
                    const nr = cur.r + dirs[i][0], nc = cur.c + dirs[i][1];
                    const nTurns = (cur.dir !== -1 && cur.dir !== i) ? cur.turns + 1 : cur.turns;
                    if (nTurns > 2 || nr < 0 || nr >= ROWS + 2 || nc < 0 || nc >= COLS + 2) continue;
                    if (board[nr][nc] === null || (nr === r2 && nc === c2)) {
                        const key = `${nr},${nc},${i},${nTurns}`;
                        if (!visited.has(key) || visited.get(key) > nTurns) {
                            visited.set(key, nTurns);
                            queue.push({ r: nr, c: nc, dir: i, turns: nTurns, path: [...cur.path, {r: nr, c: nc}] });
                        }
                    }
                }
            }
            return null;
        }

        function handleWin() {
            gameActive = false; clearInterval(timerId);
            winModal.classList.remove('hidden');
            const winTitle = document.getElementById('win-title');
            const winMessage = document.getElementById('win-message');
            const winIcon = document.getElementById('win-icon');
            
            if (currentLevel === 1) {
                winIcon.textContent = "⭐️";
                winTitle.textContent = "Уровень 1";
                winMessage.textContent = "Молодец, Аяулым! Давай дальше ✨";
                nextBtn.textContent = "Второй уровень";
                nextBtn.onclick = () => { winModal.classList.add('hidden'); currentLevel = 2; initGame(); };
            } else if (currentLevel === 2) {
                winIcon.textContent = "🚀";
                winTitle.textContent = "Уровень 2";
                winMessage.textContent = "СИИИГМА! Еще один раунд! ❤️";
                nextBtn.textContent = "К поздравлению!";
                nextBtn.onclick = () => { winModal.classList.add('hidden'); currentLevel = 3; initGame(); };
            } else if (currentLevel === 3) {
                winIcon.textContent = "💐";
                winTitle.textContent = "С Праздником!";
                winMessage.textContent = "Дорогая Аяулым, поздравляю тебя с 8 марта и желаю тебе всего самого наилучшего. Спасибо за то что ты есть ❤️";
                nextBtn.textContent = "Секретный уровень 🤫";
                nextBtn.onclick = () => { winModal.classList.add('hidden'); currentLevel = 4; initGame(); };
            } else {
                winIcon.textContent = "👑";
                winTitle.textContent = "Антистресс Мастер!";
                winMessage.textContent = "Ты прошла всё! Ты просто чудо. С праздником еще раз! 🎉";
                nextBtn.textContent = "Играть снова";
                nextBtn.onclick = () => location.reload();
            }
        }

        function drawConnectLine(path) {
            resizeCanvas();
            ctx.beginPath(); ctx.strokeStyle = '#fb7185'; ctx.lineWidth = 6;
            ctx.lineCap = 'round'; ctx.lineJoin = 'round';
            path.forEach((p, i) => {
                const {x, y} = getCellCenter(p.c, p.r);
                if (i === 0) ctx.moveTo(x, y); else ctx.lineTo(x, y);
            });
            ctx.stroke();
        }

        function getCellCenter(c, r) {
            const rect = boardElement.getBoundingClientRect();
            return { x: (c - 0.5) * (rect.width / COLS), y: (r - 0.5) * (rect.height / ROWS) };
        }

        function resizeCanvas() {
            const rect = boardElement.getBoundingClientRect();
            canvas.width = rect.width; canvas.height = rect.height;
        }

        function clearCanvas() { ctx.clearRect(0, 0, canvas.width, canvas.height); }
        function checkWin() { return !board.some(row => row.some(cell => cell !== null)); }

        document.getElementById('hint-btn').onclick = () => {
            if (!gameActive || hintsLeft <= 0 || isProcessing) return;
            const move = checkAnyPossibleMove();
            if (move) {
                hintsLeft--;
                document.getElementById('hints-count').textContent = hintsLeft;
                const el1 = document.querySelector(`[data-r="${move.t1.r}"][data-c="${move.t1.c}"]`);
                const el2 = document.querySelector(`[data-r="${move.t2.r}"][data-c="${move.t2.c}"]`);
                el1.classList.add('hint-highlight'); el2.classList.add('hint-highlight');
                setTimeout(() => { el1.classList.remove('hint-highlight'); el2.classList.remove('hint-highlight'); }, 1500);
            }
        };

        document.getElementById('shuffle-btn').onclick = () => { if (gameActive && !isProcessing) {
            let tiles = [];
            for (let r = 1; r <= ROWS; r++) for (let c = 1; c <= COLS; c++) if (board[r][c]) tiles.push(board[r][c]);
            tiles.sort(() => Math.random() - 0.5);
            for (let r = 1; r <= ROWS; r++) for (let c = 1; c <= COLS; c++) if (board[r][c]) board[r][c] = tiles.pop();
            ensurePossibleMoves(); renderBoard();
        }};

        document.getElementById('restart-btn').onclick = () => { currentLevel = 1; initGame(); };
        window.onresize = resizeCanvas;
        startApp();
    </script>
